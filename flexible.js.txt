1.基础使用
	元素尺寸/设计稿尺寸/10 rem
2.字体大小
	字体	大小建议使用px
	div {
	    width: 1rem; 
	    height: 0.4rem;
	    font-size: 12px; // 默认写上dpr为1的fontSize
	}
	[data-dpr="2"] div {
	    font-size: 24px;
	}
	[data-dpr="3"] div {
	    font-size: 36px;
	}
	

	@mixin font-dpr($font-size){
	    font-size: $font-size;
	
	    [data-dpr="2"] & {
	        font-size: $font-size * 2;
	    }
	
	    [data-dpr="3"] & {
	        font-size: $font-size * 3;
	    }
	}
